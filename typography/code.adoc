== Inline

== Block

.No source highlight
----
$ echo 'hi'
----

.Simple YAML
[source,yaml]
----
tolerations:
- key: "key1"
  operator: "Equal"
  value: "value1"
  effect: "NoExecute"
  tolerationSeconds: 3600
----

:plugin: from-an-attribute

.Mixed YAML with attribute
[source,yaml,subs="attributes+"]
----
apiVersion: operator.openshift.io/v1
kind: Network
metadata:
  name: cluster
spec:
  additionalNetworks: <1>
  - name: test-network-1
    namespace: test-1
    type: Raw
    rawCNIConfig: '{
      "name": "test-network-1",
      "type": "{plugin}",
      "master": "eth1",
      "mode": "l2",
      "ipam": {
        "type": "dhcp"
      }
    }'
----
<1> This is a callout.
